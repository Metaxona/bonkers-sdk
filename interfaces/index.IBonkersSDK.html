Interface IBonkersSDK

Interface that ensures all important SDK methods are being implemented allowing access to all contract interacting class like Controller, Vault, and Vault Factory using a single mode and config

interface IBonkersSDK {
    config: Config;
    mode: Mode;
    account(): Readonly<undefined | `0x${string}`>;
    chain(): ChainReturnType;
    chains(): readonly Chain[];
    connect(connector: Connector, callback?: ConnectCallback): Promise<ConnectReturnType<Config>>;
    connection(): Readonly<ConnectionState>;
    connectors(): readonly Connector[];
    controller(controllerParams?: BaseParams): IController;
    disconnect(): Promise<void>;
    erc20(tokenAddress?: `0x${string}`): IErc20;
    getContractType(address: `0x${string}`): Promise<ContractType>;
    getContractVersion(address: `0x${string}`): Promise<string>;
    reader(params: ReaderParams): Promise<unknown>;
    reconnect(connectors: Connector[]): Promise<ReconnectReturnType>;
    switchAccount(connector: Connector, callback?: SwitchAccountCallback): Promise<SwitchAccountReturnType>;
    switchChain(chainId: number, callback?: SwitchChainCallback): Promise<Chain>;
    useAccount(privateKey: `0x${string}`): void;
    useChain(chainId: number): void;
    vault(vaultParams?: BaseParams): IVault;
    vaultFactory(vaultFactoryParams?: BaseParams): IVaultFactory;
    writer(params: WriterParams): Promise<ContractInteractionReturnType<`0x${string}`>>;
}
